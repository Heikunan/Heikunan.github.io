# CSS 布局

# 左右布局

html 代码

<div class="container">
  <div class="containerLeft"></div>
  <div class="containerRight"></div>
</div>

css 代码

.container {
  display: flex;
  flex-direction: row;
}
.containerLeft {
  flex: 1;
}
.containerRight {
  flex: 1;
}

# 左中右布局

html 代码

<div class="container">
  <div class="containerLeft"></div>
  <div class="containerCenter"></div>
  <div class="containerRight"></div>
</div>

css 代码

.container {
  display: flex;
  flex-direction: row;
}
.containerLeft {
  flex: 1;
}
.containerCenter {
  width: 200px;
}
.containerRight {
  flex: 1;
}

# 水平居中

html 代码

<div class="container"><div class="containerCenter"></div></div>

css 代码

.container {
  display: flex;
  flex-direction: row;
}

.containerCenter {
  width: 200px;
  justify-content: center;
}

# 垂直居中

<div class="container"><div class="containerCenter"></div></div>

css 代码

.container {
  display: flex;
  flex-direction: row;
}

.containerCenter {
  width: 200px;
  height: 200px;
  align-content: center;
}
