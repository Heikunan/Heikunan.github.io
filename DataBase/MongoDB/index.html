# 在 CentOS 里安装 MongoDB

# 使用系统命令(yum)安装

1.打开/etc/yum.repos.d 文件加

cd /etc/yum.repos.d touch mongodb-org-3.6.repo

2.编辑 mongodb-org-3.6.repo 文件

vi mongodb-org-3.6.repo

//将以下配置写入文件内

[mongodb-org-3.6]
name=MongoDB Repository
baseurl=https://repo.mongodb.org/yum/redhat/$releasever/mongodb-org/3.6/x86_64/
gpgcheck=1
enabled=1
gpgkey=https://www.mongodb.org/static/pgp/server-3.6.asc

3.直接使用 yum 命令安装 MongoDB

yum install -y mongodb-org

# 配置 MongoDB 可以远程访问

1.先修改 mongo.conf 文件

vi /etc/mongod.conf
//找到bindIp修改为你自己的IP
bindIp: xx.xxx.xx.xx(你自己服务器的IP)

2.重启 MongoDB 应用

service mongod restart

3.在查看自己端口的使用情况

netstat -nputl
//如果出现以下情况即为成功
//xx.xxx.xx.xx为你服务器的IP地址
xx.xxx.xx.xx:27017  /mongod
